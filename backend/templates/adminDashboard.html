Total Waste Collected Per Area

SELECT a.name AS area_name, 
       SUM(w.bio_wt) AS total_bio_waste, 
       SUM(w.non_bio_wt) AS total_non_bio_waste
FROM waste w
JOIN user u ON w.user_id = u.user_id
JOIN area a ON u.area_id = a.area_id
GROUP BY a.area_id
ORDER BY (SUM(w.bio_wt) + SUM(w.non_bio_wt)) DESC;

Drivers Who Have Collected the Most Waste

SELECT v.driver_name, v.vehicle_id, 
       SUM(w.bio_wt + w.non_bio_wt) AS total_waste_collected
FROM waste w
JOIN vehicle v ON w.vehicle_id = v.vehicle_id
WHERE v.driver_name IS NOT NULL
GROUP BY v.vehicle_id, v.driver_name
ORDER BY total_waste_collected DESC
LIMIT 10;

Vehicles That Have Not Collected Waste Recently

SELECT v.vehicle_id, v.driver_name, v.driver_phone, v.license_plate, v.type,
       MAX(w.c_date_time) AS last_collection_date
FROM vehicle v
LEFT JOIN waste w ON v.vehicle_id = w.vehicle_id
GROUP BY v.vehicle_id, v.driver_name, v.driver_phone, v.license_plate, v.type
HAVING last_collection_date IS NULL OR last_collection_date < date('now', '-30 day')
ORDER BY last_collection_date;

Pending Complaints and Their Users

SELECT c.complaint_id, c.message, c.comp_date, 
       u.first_name || ' ' || u.last_name AS user_name,
       u.email, a.name AS area_name
FROM complaint c
JOIN user u ON c.user_id = u.user_id
JOIN area a ON c.area_id = a.area_id
WHERE c.status = 'Pending' OR c.status IS NULL
ORDER BY c.comp_date ASC;

Areas with Most Complaints

SELECT a.name AS area_name, COUNT(c.complaint_id) AS total_complaints,
       MAX(c.comp_date) AS latest_complaint_date
FROM complaint c
JOIN area a ON c.area_id = a.area_id
GROUP BY a.area_id, a.name
ORDER BY total_complaints DESC;

Waste Collections with Pending Rewards

SELECT w.waste_id, u.first_name || ' ' || u.last_name AS user_name,
       w.c_date_time, w.bio_wt, w.non_bio_wt, w.waste_tag,
       w.reward_status
FROM waste w
JOIN user u ON w.user_id = u.user_id

Average Waste Collection Per Vehicle

SELECT v.vehicle_id, v.driver_name, v.type, v.license_plate,
       COUNT(w.waste_id) AS collection_count,
       AVG(w.bio_wt + w.non_bio_wt) AS avg_waste_per_collection,
       SUM(w.bio_wt + w.non_bio_wt) AS total_waste_collected
FROM waste w
JOIN vehicle v ON w.vehicle_id = v.vehicle_id
GROUP BY v.vehicle_id, v.driver_name, v.type, v.license_plate
ORDER BY avg_waste_per_collection DESC;

Most Efficient Areas in Waste Collection

SELECT a.name AS area_name, COUNT(w.waste_id) AS collections_count, 
       SUM(w.bio_wt) AS total_bio_waste,
       SUM(w.non_bio_wt) AS total_non_bio_waste,
       SUM(w.bio_wt + w.non_bio_wt) AS total_waste_collected
FROM waste w
JOIN user u ON w.user_id = u.user_id
JOIN area a ON u.area_id = a.area_id
GROUP BY a.area_id, a.name
ORDER BY collections_count DESC;
